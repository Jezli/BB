'text-green-400' : 'text-red-400'; portfolioHtml += `
${asset.label} 现价: $${currentPrice.toLocaleString()}

初始投入 (HKD)

≈ ${initialInvestHkd.toLocaleString(undefined,{maximumFractionDigits:2})}

买入单价

$${asset.buyPrice}

当前价值 (HKD)

≈ ${currentValHkd.toLocaleString(undefined,{maximumFractionDigits:2})}

累计盈亏

${profitLossHkd >= 0 ? '+' : ''}${profitLossHkd.toFixed(2)} HKD

`; }); // ... 大盘和总额计算逻辑保持不变 ... // (此处省略部分代码，直接在你的原代码基础上应用上述卡片逻辑即可) document.getElementById('portfolio-list').innerHTML = portfolioHtml; // ... 剩余代码 ... } catch (e) { console.log("连接中..."); } }
